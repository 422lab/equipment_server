001package com.kystu.equipment.model;
002
003import com.dxzc.json.ObjectGen;
004import com.kystu.equipment.Tools;
005import com.mysql.cj.xdevapi.DbDoc;
006import com.mysql.cj.xdevapi.JsonNumber;
007import com.mysql.cj.xdevapi.JsonValue;
008
009import java.sql.Timestamp;
010import java.util.Map;
011
012public class DeviceReserve {
013
014    public Timestamp start;
015    public Timestamp end;
016    public int user;
017
018
019    public void fromJson(JsonValue json) {
020        DbDoc map = (DbDoc) json;
021        for (Map.Entry<String, JsonValue> entry : map.entrySet()) {
022            switch (entry.getKey()) {
023                case "start":
024                    start = Tools.timestamp((JsonNumber) entry.getValue());
025                    break;
026                case "end":
027                    end = Tools.timestamp((JsonNumber) entry.getValue());
028                    break;
029                case "user":
030                    user = ((JsonNumber) entry.getValue()).getInteger();
031                    break;
032            }
033        }
034    }
035
036    public void toJson(ObjectGen json) {
037        json.number("start", start.getTime());
038        json.number("end", end.getTime());
039        json.number("user", user);
040    }
041}
