001package com.kystu.equipment.servlet;
002
003import com.dxzc.json.ArrayGen;
004import com.dxzc.json.ObjectGen;
005import com.kystu.equipment.GetPostTools;
006import com.kystu.equipment.Tools;
007import com.kystu.equipment.dao.DeviceDao;
008import com.kystu.equipment.dao.UserDao;
009import com.kystu.equipment.model.*;
010import com.mysql.cj.xdevapi.DbDoc;
011import com.mysql.cj.xdevapi.JsonParser;
012
013import javax.servlet.ServletException;
014import javax.servlet.annotation.WebServlet;
015import javax.servlet.http.HttpServletRequest;
016import javax.servlet.http.HttpServletResponse;
017import java.io.IOException;
018import java.io.StringReader;
019import java.sql.Connection;
020import java.sql.SQLException;
021import java.sql.Timestamp;
022import java.util.*;
023
024@WebServlet(name = "UserReserveDevice", value = "/userReserveDevice")
025public class UserReserveDeviceServlet extends BaseServlet {
026
027    /**
028     * 登录成功返回码.
029     */
030    public static final int SUCCESS = 0;
031
032    /**
033     * 登录失败返回码.
034     */
035    public static final int ERROR = 1;
036
037    /**
038     * 登录参数错误返回码.
039     */
040    public static final int PARAMETER = 2;
041
042    @Override
043    protected void doGetOrPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
044        resp.setContentType("application/json;charset=utf-8");
045        GetPostTools tools = new GetPostTools(req);
046        ObjectGen json = new ObjectGen();
047        Timestamp now = Tools.currentTimestamp();
048        try {
049            String account = tools.getParameter("account");
050            String password = tools.getParameter("password");
051            if (account == null || password == null) {
052                json.number("code", 1);
053                json.string("msg", "account or password");
054                return;
055            }
056            UseReserve reserve;
057            try {
058                reserve = new UseReserve(JsonParser.parseDoc(tools.getParameter("reserves")));
059                if (reserve.start.before(now)) {
060                    reserve.start = now;
061                }
062            } catch (RuntimeException e) {
063                json.number("code", 1);
064                json.string("msg", "json?" + e.toString());
065                return;
066            }
067            Calendar calendar = Calendar.getInstance();
068            calendar.setTime(reserve.start);
069            calendar.add(Calendar.MINUTE, 5);
070            if (calendar.getTime().after(reserve.end)) {
071                json.number("code", 1);
072                json.string("msg", "reserve time");
073                return;
074            }
075            calendar.setTime(now);
076            calendar.add(Calendar.HOUR, 12);
077            Timestamp dead = new Timestamp(calendar.getTime().getTime());
078            if (dead.before(reserve.end)) {
079                reserve.end = dead;
080            }
081
082            if (reserve.devices.isEmpty()) {
083                json.number("code", 1);
084                json.string("msg", "reserve device");
085                return;
086            }
087
088            try (Connection connection = Tools.getConnection()) {
089                connection.setAutoCommit(false);
090                UserDao userDao = new UserDao(connection);
091                DeviceDao deviceDao = new DeviceDao(connection);
092
093                // 执行用户登录
094                User user = userDao.getUserForUpdate(account);
095                if (user == null) {
096                    json.number("code", 2);
097                    return;
098                }
099                if (!password.equals(user.password)) {
100                    json.number("code", 3);
101                    return;
102                }
103
104                // 执行特点用户类型的相关限制检查
105                switch (user.type) {
106                    case 0: {
107                        // 学生一次只能预约一台设备
108                        if (reserve.devices.size() > 1) {
109                            json.number("code", 4);
110                            return;
111                        }
112                        // 学生同类型只能预约一种设备
113                        if (!user.getReserve().checkOne(reserve.type, reserve.start, reserve.end)) {
114                            json.number("code", 4);
115                            return;
116                        }
117                        break;
118                    }
119                    default:
120                        throw new RuntimeException("未知用户类型");
121                }
122
123                List<Integer> success = new LinkedList<>();
124                List<Integer> fails = new LinkedList<>();
125                // 检查全部预约设备可用性
126                // 避免死锁
127                for (Iterator<Integer> it = reserve.devices.iterator(); it.hasNext(); ) {
128                    int deviceUuid = it.next();
129                    Device device = deviceDao.getDeviceForUpdate(deviceUuid);
130                    // 检查设备类型和可用性
131                    if (device.type != reserve.type || !device.check(reserve.start, reserve.end)) {
132                        // 不满足就移除
133                        it.remove();
134                        // 添加失败项
135                        fails.add(deviceUuid);
136                    } else {
137                        DeviceReserve deviceReserve = new DeviceReserve();
138                        deviceReserve.user = user.uuid;
139                        deviceReserve.start = reserve.start;
140                        deviceReserve.end = reserve.end;
141                        device.reserves.get().add(deviceReserve);
142                        deviceDao.updateDevice(device);
143                        connection.commit();
144                        // 添加成功项
145                        success.add(deviceUuid);
146                    }
147                }
148                user.removeBefore(now);
149                user.getReserve().add(reserve);
150                userDao.updateUser(user);
151                connection.commit();
152                json.number("code", 0);
153                try (ArrayGen array = json.array("fails")) {
154                    for (int f : fails) {
155                        array.number(f);
156                    }
157                }
158                try (ArrayGen array = json.array("success")) {
159                    for (int f : success) {
160                        array.number(f);
161                    }
162                }
163                json.number("start", reserve.start.getTime());
164                json.number("end", reserve.end.getTime());
165            } catch (SQLException e) {
166                throw new RuntimeException(e);
167            }
168        } finally {
169            resp.getWriter().println(json.toJson());
170        }
171    }
172}
