001package com.kystu.equipment;
002
003import java.sql.Timestamp;
004
005/**
006 * 生成JSON的辅助代码.
007 */
008public class Json {
009
010    /**
011     * 对字符串进行转义.
012     *
013     * @param str 需要转义的字符串
014     * @return 转义结果
015     */
016    public static String json(String str) {
017        if (str == null) {
018            return "null";
019        }
020        StringBuilder sb = new StringBuilder();
021        sb.append("\"");
022        for (int i = 0; i < str.length(); i++) {
023            char c = str.charAt(i);
024            switch (c) {
025                case '\r':
026                    sb.append("\\r");
027                    break;
028                case '\n':
029                    sb.append("\\n");
030                    break;
031                case '\t':
032                    sb.append("\\t");
033                    break;
034                case '\"':
035                    sb.append("\\\"");
036                    break;
037                case '\'':
038                    sb.append("\\'");
039                    break;
040                default:
041                    sb.append(c);
042            }
043        }
044        sb.append("\"");
045        return sb.toString();
046    }
047
048    /**
049     * 生成时间戳json表示.
050     *
051     * @param time 时间戳
052     * @return JSON表示
053     */
054    public static String json(Timestamp time) {
055        if (time == null) {
056            return "null";
057        }
058        return String.valueOf(time.getTime());
059    }
060
061}
