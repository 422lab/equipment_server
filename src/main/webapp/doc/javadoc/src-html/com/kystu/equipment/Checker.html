001package com.kystu.equipment;
002
003/**
004 * 对接口参数进行的检查项.
005 */
006public class Checker {
007
008    /**
009     * 检查账号是否合法.
010     *
011     * @param account 账号
012     * @return 合法性
013     */
014    public static boolean isValidAccount(String account) {
015        return account != null && account.matches("^\\w+$");
016    }
017
018    /**
019     * 检查密码是否合法.
020     *
021     * @param account 密码
022     * @return 合法性
023     */
024    public static boolean isValidPassword(String account) {
025        return account != null && account.matches("^\\w+$");
026    }
027
028}
