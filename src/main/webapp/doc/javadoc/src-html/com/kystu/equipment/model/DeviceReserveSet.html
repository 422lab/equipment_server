001package com.kystu.equipment.model;
002
003import com.dxzc.json.ArrayGen;
004import com.dxzc.json.ObjectGen;
005import com.kystu.equipment.Tools;
006import com.mysql.cj.xdevapi.DbDoc;
007import com.mysql.cj.xdevapi.JsonArray;
008import com.mysql.cj.xdevapi.JsonValue;
009
010import java.sql.Timestamp;
011import java.util.HashSet;
012import java.util.Iterator;
013import java.util.Map;
014import java.util.stream.Collectors;
015
016public class DeviceReserveSet implements Iterable<DeviceReserve> {
017    public DeviceReserveSet() {
018
019    }
020
021    public DeviceReserveSet(JsonValue json) {
022        DbDoc object = (DbDoc) json;
023        for (Map.Entry<String, JsonValue> entry : object.entrySet()) {
024            switch (entry.getKey()) {
025                case "set": {
026                    for (JsonValue item : (JsonArray) entry.getValue()) {
027                        DeviceReserve reserve = new DeviceReserve();
028                        reserve.fromJson(item);
029                        set.add(reserve);
030                    }
031                    break;
032                }
033            }
034        }
035    }
036
037    private final HashSet<DeviceReserve> set = new HashSet<>();
038
039    public boolean add(DeviceReserve reserve) {
040        return set.add(reserve);
041    }
042
043    public void toJson(ObjectGen json) {
044        try (ArrayGen array = json.array("set")) {
045            for (DeviceReserve reserve : set) {
046                try (ObjectGen item = array.object()) {
047                    reserve.toJson(item);
048                }
049            }
050        }
051    }
052
053    public int size() {
054        return set.size();
055    }
056
057    public boolean removeBefore(Timestamp timestamp) {
058        return set.removeIf(reserve -> reserve.end.before(timestamp));
059    }
060
061    public DeviceReserve get(Timestamp timestamp) {
062        return Tools.onlyOneOr(set.stream()
063                .filter(reserve ->
064                        reserve.start.before(timestamp) && reserve.end.after(timestamp))
065                .collect(Collectors.toSet()), null);
066    }
067
068    public boolean check(Timestamp start, Timestamp end) {
069        for (DeviceReserve i : set) {
070            if (i.end.after(start) && i.start.before(end)) {
071                return false;
072            }
073        }
074        return true;
075    }
076
077    @Override
078    public Iterator<DeviceReserve> iterator() {
079        return set.iterator();
080    }
081
082    public boolean removeReserve(Timestamp start, Timestamp end) {
083        return set.removeIf(i -> Tools.equals(i.start, start) && Tools.equals(i.end, end));
084    }
085}
