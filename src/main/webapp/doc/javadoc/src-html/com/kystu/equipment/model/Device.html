001package com.kystu.equipment.model;
002
003import com.dxzc.tools.DelayLoader;
004import com.kystu.equipment.Tools;
005
006import java.sql.Timestamp;
007import java.util.Set;
008
009public class Device {
010
011    public int uuid;
012    public String password;
013    public int type;
014    public String description;
015    public String local;
016    public DelayLoader<String, DeviceReserveSet> reserves;
017    public String control;
018    public String state;
019    public Timestamp last;
020
021    public DeviceReserveSet getReserves() {
022        return reserves.get();
023    }
024
025    public boolean reserveRemoveBefore(Timestamp timestamp) {
026        return getReserves().removeBefore(timestamp);
027    }
028
029    public int reservesSize() {
030        return getReserves().size();
031    }
032
033    public DeviceReserve getReserve(Timestamp timestamp) {
034        return getReserves().get(timestamp);
035    }
036
037    public boolean check(Timestamp start, Timestamp end) {
038        return getReserves().check(start, end);
039    }
040
041    public boolean removeReserve(Timestamp start, Timestamp end) {
042        return getReserves().removeReserve(start, end);
043    }
044}
